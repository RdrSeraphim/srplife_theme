{{- /* Renders an embedded YouTube video. @param {int} [end] The time, measured in seconds from the start of the video, when the player should stop playing the video. @param {string} [id] The video id. Optional if the id is the first and only positional argument. @param {bool} [loop=false] Whether to indefinitely repeat the video. Ignores the start and end arguments after the first play. @param {int} [start] The time, measured in seconds from the start of the video, when the player should start playing the video. @returns {template.HTML} @reference https://developers.google.com/youtube/player_parameters @example {{% yt 0RKpf3rK57I %}}Caption here.{{% /yt %}} @example {{% yt id=0RKpf3rK57I start=30 %}}Caption here.{{% /yt %}} */}} {{- $pc := .Page.Site.Config.Privacy.YouTube }} {{- if not $pc.Disable }} {{- with $id := or (.Get "id") (.Get 0) }} {{- /* Set defaults. */}} {{- $end := 0 }} {{- $loop := 0 }} {{- $start := 0 }} {{- /* Get arguments. */}} {{- if in (slice "true" true 1) ($.Get "loop") }} {{- $loop = 1 }} {{- else if in (slice "false" false 0) ($.Get "loop") }} {{- $loop = 0 }} {{- end }} {{- $end := or ($.Get "end") $end }} {{- $start := or ($.Get "start") $start }} {{- /* Define src attribute. */}} {{- $src := printf "https://www.youtube-nocookie.com/embed/%s" $id }} {{- $params := dict "feature" "oembed" "end" $end "start" $start "loop" $loop }} {{- with querify $params }} {{- $src = printf "%s?%s" $src . }} {{- end }} {{- /* Render. */ -}} {{- else }} {{- errorf "The %q shortcode requires an id argument. See %s" .Name .Position }} {{- end }} {{- end }}
